        else @hash[num_or_key]

    def []=( key, value )
        index = @ary.find { |item| item == @hash[key] }

        if @hash[key].nil? or index.nil?
            @ary << value
        else
            @ary[index] = value
        end

        @hash[key] = value
    end

    def resolve( task_name )
        task = self[task_name]

        if task.nil? then nil
        else task.resolve(self)
        end
    end
